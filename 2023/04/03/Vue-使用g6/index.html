        // 给g6的graph实例，添加edge类型的边

        // 将边对象，push到data对象中
        data.edges.push(modelEdge)

        for (let i = data.nodes.length - 1; i > g6_id.value; i--) {
            data.nodes[i].label = data.nodes[i - 1].label;
