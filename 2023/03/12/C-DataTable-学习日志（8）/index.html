读取csv实例

根据特点，它可以用于读取csv文件中的内容，并存储在dataTable中.
如果直接使用CsvHelper，那样是存储在一个List<Object>中
以下是例子： 

using System;
using System.Data;
using System.IO;
using System.Text;
using System.Diagnostics;

namespace Nestar.Helpers
{
    public class SpreadSheetHelper
    {

        /// <summary>
        /// 将DataTable中数据写入到表格文件中
        /// </summary>
        /// <param name="Table">源数据的DataTable</param>
        /// <param name="FullPath">文件路径</param>
        /// <param name="LineSeparator">行分隔符</param>
        /// <param name="FieldDelimiter">列分隔符</param>
        public static void SaveDataTableToSpreadsheet(DataTable Table, string FullPath, string LineSeparator, string FieldDelimiter )
        {
            FileInfo fileInfo = new FileInfo(FullPath);
            if (!fileInfo.Directory.Exists)
            {
                fileInfo.Directory.Create();
            }
            using (FileStream fs = new FileStream(FullPath, FileMode.Create, FileAccess.Write))
            {
                using (StreamWriter sw = new StreamWriter(fs, Encoding.Default))
                {
                    string data = "";
                    //写出列名称
                    for (int i = 0; i < Table.Columns.Count; i++)
                    {
                        data += Table.Columns[i].ColumnName.ToString();
                        if (i < Table.Columns.Count - 1)
                        {
                            data += FieldDelimiter;
                        }
                    }
                    sw.Write(data + LineSeparator);

                    //写出各行数据
                    for (int i = 0; i < Table.Rows.Count; i++)
                    {
                        data = "";
                        bool IsEmptyLine = true;
                        for (int j = 0; j < Table.Columns.Count; j++)
                        {
                            IsEmptyLine = IsEmptyLine && string.IsNullOrEmpty(Table.Rows[i][j].ToString());
                            data += Table.Rows[i][j].ToString();
                            if (j < Table.Columns.Count - 1)
                            {
                                data += FieldDelimiter;
                            }
                        }
                        if (!IsEmptyLine)
                        {
                            sw.Write(data + LineSeparator);
                        }

                    }
                    sw.Flush();
                    sw.Close();
                }
                fs.Close();
            }
        }

        /// <summary>
        /// 将DataTable中数据写入到CSV文件中
        /// </summary>
        /// <param name="Table">源数据的DataTable</param>
        /// <param name="FullPath">CSV文件路径</param>
        public static void SaveDataTableToCSV(DataTable Table, string FullPath)
        {
            SaveDataTableToSpreadsheet(Table, FullPath, "\r\n", ",");
        }


        /// <summary>
        /// 将表格文件的数据读取到DataTable中
        /// </summary>
        /// <param name="FilePath">表格文件路径</param>
        /// <returns>返回读取了表格数据的DataTable</returns>
        /// <param name="LineSeparator">行分隔符</param>
        /// <param name="FieldDelimiter">列分隔符</param>
        public static DataTable OpenSpreadsheetAsDataTable(string FilePath, string LineSeparator, string FieldDelimiter)
        {
            DataTable dataTable = new DataTable();
            if (!string.IsNullOrEmpty(FilePath))
            {
                Encoding encoding = GetType(FilePath);
                using (FileStream fs = new FileStream(FilePath, FileMode.Open, FileAccess.Read))
                {

                    using (StreamReader sr = new StreamReader(fs, encoding))
                    {
                        string fileContent = sr.ReadToEnd();
                        string[] strLines = fileContent.Split(new string[] { LineSeparator }, StringSplitOptions.RemoveEmptyEntries);

                        //记录每行记录中的各字段内容
                        string[] tableHead = null;
                        string[] fields = null;
                        //标示列数
                        int columnCount = 0;
                        //标示是否是读取的第一行
                        bool isFirstLine = true;
                        try
                        {
                            foreach (string sline in strLines)
                            {
                                if (!string.IsNullOrEmpty(sline))
                                {
                                    if (isFirstLine)
                                    {
                                        isFirstLine = !isFirstLine;
                                        tableHead = sline.Split(new string[] { FieldDelimiter }, StringSplitOptions.RemoveEmptyEntries);
                                        columnCount = tableHead.Length;
                                        foreach (string head in tableHead)
                                        {
                                            dataTable.Columns.Add(head);
                                        }
                                    }
                                    else
                                    {
                                        fields = sline.Split(new string[] { FieldDelimiter },StringSplitOptions.None);
                                        DataRow dr = dataTable.NewRow();
                                        for (int j = 0; j < columnCount; j++)
                                        {
                                            dr[j] = fields[j];
                                        }
                                        dataTable.Rows.Add(dr);
                                    }
                                }
                            }
                        }
                        catch
                        { }
                        sr.Close();
                    }
                    fs.Close();
                }
            }
            return dataTable;
        }

        /// <summary>
        /// 将CSV文件的数据读取到DataTable中
        /// </summary>
        /// <param name="FilePath">CSV文件路径</param>
        /// <returns>返回读取了CSV数据的DataTable</returns>
        public static DataTable OpenCSVAsDataTable(string FilePath)
        {
            return OpenSpreadsheetAsDataTable(FilePath, "\r\n", ",");
        }

        public static bool AppendToCSV(string FilePath, System.Collections.IEnumerable Fields)
        {
            bool retVal;
            try
            {
                string dir = Path.GetDirectoryName(FilePath);
                if (!Directory.Exists(dir))
                {
                    Directory.CreateDirectory(dir);
                }
                using (FileStream fs = new FileStream(FilePath, FileMode.Append, FileAccess.Write))
                {
                    Debug.Print(fs.Length.ToString());
                    using (StreamWriter sw = new StreamWriter(fs))
                    {
                        string context = string.Empty;
                        foreach (object field in Fields)
                        {
                            context += field.ToString() + ",";
                        }
                        context = context.TrimEnd(',');
                        sw.Write(context + "\r\n");
                        sw.Close();
                    }
                    fs.Close();
                }
                retVal = true;
            }
            catch
            {
                retVal = false;
            }
            return retVal;
        }

        /// 给定文件的路径，读取文件的二进制数据，判断文件的编码类型  
        /// <param name="FileName">文件路径</param>  
        /// <returns>文件的编码类型</returns>  
        public static Encoding GetType(string FileName)
        {
            FileStream fs = new FileStream(FileName, FileMode.Open, FileAccess.Read);
            Encoding r = GetType(fs);
            fs.Close();
            return r;
        }

        /// 通过给定的文件流，判断文件的编码类型  
        /// <param name="fileStream">文件流</param>  
        /// <returns>文件的编码类型</returns>  
        public static Encoding GetType(FileStream fileStream)
        {
            byte[] Unicode = new byte[] { 0xFF, 0xFE, 0x41 };
            byte[] UnicodeBIG = new byte[] { 0xFE, 0xFF, 0x00 };
            byte[] UTF8 = new byte[] { 0xEF, 0xBB, 0xBF }; //带BOM  
            Encoding retVal = Encoding.Default;

            BinaryReader r = new BinaryReader(fileStream, Encoding.Default);
            int.TryParse(fileStream.Length.ToString(), out int i);
            byte[] ss = r.ReadBytes(i);
            if (IsUTF8Bytes(ss) || (ss[0] == 0xEF && ss[1] == 0xBB && ss[2] == 0xBF))
            {
                retVal = Encoding.UTF8;
            }
            else if (ss[0] == 0xFE && ss[1] == 0xFF && ss[2] == 0x00)
            {
                retVal = Encoding.BigEndianUnicode;
            }
            else if (ss[0] == 0xFF && ss[1] == 0xFE && ss[2] == 0x41)
            {
                retVal = Encoding.Unicode;
            }
            r.Close();
            return retVal;
        }

        /// 判断是否是不带 BOM 的 UTF8 格式  
        /// <param name="data"></param>  
        /// <returns></returns>  
        private static bool IsUTF8Bytes(byte[] data)
        {
            int charByteCounter = 1;  //计算当前正分析的字符应还有的字节数  
            byte curByte; //当前分析的字节.  
            for (int i = 0; i < data.Length; i++)
            {
                curByte = data[i];
                if (charByteCounter == 1)
                {
                    if (curByte >= 0x80)
                    {
                        //判断当前  
                        while (((curByte <<= 1) & 0x80) != 0)
                        {
                            charByteCounter++;
                        }
                        //标记位首位若为非0 则至少以2个1开始 如:110XXXXX...........1111110X　  
                        if (charByteCounter == 1 || charByteCounter > 6)
                        {
                            return false;
                        }
                    }
                }
                else
                {
                    //若是UTF-8 此时第一位必须为1  
                    if ((curByte & 0xC0) != 0x80)
                    {
                        return false;
                    }
                    charByteCounter--;
                }
            }
            if (charByteCounter > 1)
            {
                throw new Exception("非预期的byte格式");
            }
            return true;
        }
        
    }
}
